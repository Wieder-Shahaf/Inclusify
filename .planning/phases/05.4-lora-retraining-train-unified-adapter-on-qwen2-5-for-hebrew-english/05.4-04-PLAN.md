**Modify infra/azure/vllm-vm/vllm.service:**

Add `--lora-modules` and `--enable-lora` flags to ExecStart:

```ini
[Unit]
Description=vLLM OpenAI-compatible API Server
After=network.target

[Service]
Type=simple
User=azureuser
WorkingDirectory=/home/azureuser
Environment="PATH=/home/azureuser/vllm-venv/bin:/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin"
ExecStart=/home/azureuser/vllm-venv/bin/python -m vllm.entrypoints.openai.api_server \
    --model /home/azureuser/models/Qwen2.5-3B-Instruct-GPTQ-Int4 \
    --host 127.0.0.1 \
    --port 8001 \
    --dtype half \
    --max-model-len 2048 \
    --enable-lora \
    --lora-modules inclusify=/home/azureuser/inclusify/ml/adapters/best_config
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target
```

**Key changes:**
- Add `--enable-lora` flag (enables LoRA support in vLLM)
- Add `--lora-modules inclusify=/path/to/adapter/` (names adapter "inclusify")
- Path points to symlink created in Plan 03 (best_config -> qwen_r16_d0.1/)

**Why this approach:**
- vLLM 0.6.6 supports LoRA adapters via --lora-modules flag
- Adapter name "inclusify" used in OpenAI API requests: `model: inclusify`
- Symlink approach allows updating adapter without changing service file

**Create scripts/test-vllm-adapter.sh:**
```bash
#!/bin/bash
#
# Test vLLM LoRA adapter inference
# Verifies adapter produces classification output for LGBTQ+ language analysis

set -e

VM_IP="52.224.246.238"
ENDPOINT="http://localhost:8001/v1/chat/completions"

echo "=== Inclusify vLLM LoRA Adapter Test ==="
echo "VM IP: $VM_IP"
echo ""

# Test 1: Check /v1/models endpoint includes adapter
echo "[1/4] Checking /v1/models endpoint for adapter..."
ssh azureuser@$VM_IP "curl -s http://localhost:8001/v1/models" > /tmp/vllm_models.json

if grep -q "inclusify" /tmp/vllm_models.json; then
    echo "  PASS: 'inclusify' adapter found in model list"
else
    echo "  FAIL: 'inclusify' adapter NOT found"
    exit 1
fi

# Test 2: Classification query - Correct sentence
echo "[2/4] Testing Correct sentence classification..."
RESPONSE=$(ssh azureuser@$VM_IP "curl -s http://localhost:8001/v1/chat/completions \
    -H 'Content-Type: application/json' \
    -d '{
        \"model\": \"inclusify\",
        \"messages\": [
            {\"role\": \"system\", \"content\": \"You are an LGBTQ+ language analyzer. Classify severity: Correct, Outdated, Biased, Potentially Offensive, or Factually Incorrect.\"},
            {\"role\": \"user\", \"content\": \"Sexual orientation is not considered a mental disorder by modern psychiatry.\"}
        ],
        \"max_tokens\": 100,
        \"temperature\": 0.1
    }'" | jq -r '.choices[0].message.content')

if echo "$RESPONSE" | grep -iq "Correct"; then
    echo "  PASS: Response contains 'Correct' classification"
else
    echo "  WARNING: Response may not contain expected classification"
    echo "  Response: $RESPONSE"
fi

# Test 3: Classification query - Biased sentence
echo "[3/4] Testing Biased sentence classification..."
RESPONSE=$(ssh azureuser@$VM_IP "curl -s http://localhost:8001/v1/chat/completions \
    -H 'Content-Type: application/json' \
    -d '{
        \"model\": \"inclusify\",
        \"messages\": [
            {\"role\": \"system\", \"content\": \"You are an LGBTQ+ language analyzer. Classify severity: Correct, Outdated, Biased, Potentially Offensive, or Factually Incorrect.\"},
            {\"role\": \"user\", \"content\": \"Gender diversity is being pushed by ideological groups.\"}
        ],
        \"max_tokens\": 100,
        \"temperature\": 0.1
    }'" | jq -r '.choices[0].message.content')

if echo "$RESPONSE" | grep -Eiq "Biased|Offensive|Incorrect"; then
    echo "  PASS: Response identifies problematic language"
else
    echo "  WARNING: Response may not identify issue"
    echo "  Response: $RESPONSE"
fi

# Test 4: Verify adapter affects output vs base model
echo "[4/4] Comparing base model vs adapter output..."
BASE_RESPONSE=$(ssh azureuser@$VM_IP "curl -s http://localhost:8001/v1/chat/completions \
    -H 'Content-Type: application/json' \
    -d '{
        \"model\": \"/home/azureuser/models/Qwen2.5-3B-Instruct-GPTQ-Int4\",
        \"messages\": [
            {\"role\": \"user\", \"content\": \"Hello\"}
        ],
        \"max_tokens\": 20
    }'" | jq -r '.choices[0].message.content')

ADAPTER_RESPONSE=$(ssh azureuser@$VM_IP "curl -s http://localhost:8001/v1/chat/completions \
    -H 'Content-Type: application/json' \
    -d '{
        \"model\": \"inclusify\",
        \"messages\": [
            {\"role\": \"system\", \"content\": \"You are an LGBTQ+ language analyzer.\"},
            {\"role\": \"user\", \"content\": \"Hello\"}
        ],
        \"max_tokens\": 20
    }'" | jq -r '.choices[0].message.content')

if [ "$BASE_RESPONSE" != "$ADAPTER_RESPONSE" ]; then
    echo "  PASS: Adapter modifies model behavior"
else
    echo "  WARNING: Adapter may not be affecting output"
fi

echo ""
echo "=== Results ==="
echo "Adapter integration test complete"
echo "✅ vLLM serving 'inclusify' model with trained LoRA adapter"
```
