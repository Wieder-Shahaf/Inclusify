```bash
# SSH to VM
ssh azureuser@$VM_IP

# Create vLLM venv (separate from existing .venv to avoid conflicts)
cd /home/azureuser
python3.11 -m venv vllm-venv
source vllm-venv/bin/activate

# Install vLLM 0.6.6 (pinned - newer versions need CUDA 12.9+)
pip install --upgrade pip
pip install vllm==0.6.6 huggingface-hub

# Create models directory
mkdir -p /home/azureuser/models

# Download GPTQ model (NOT AWQ - T4 incompatible)
# This takes 5-10 minutes for ~2GB download
huggingface-cli download Qwen/Qwen2.5-3B-Instruct-GPTQ-Int4 \
    --local-dir /home/azureuser/models/Qwen2.5-3B-Instruct-GPTQ-Int4

ls -la /home/azureuser/models/Qwen2.5-3B-Instruct-GPTQ-Int4/
```

4. Note: Do NOT start vLLM yet - Task 2 will configure and start the systemd service.

**Why GPTQ not AWQ:** AWQ requires compute capability 8.0+ (Ampere). T4 is 7.5 (Turing). Using AWQ will crash with "Min capability: 80. Current capability: 75" error.

**Why vLLM 0.6.6:** Latest vLLM (0.17.0) requires CUDA 12.9+. VM has CUDA 12.4. Version 0.6.6 is stable with CUDA 12.4.

  <files>infra/azure/vllm-vm/vllm.service</files>
  <action>
1. Update `infra/azure/vllm-vm/vllm.service` to use `vllm serve` command:

```ini
[Unit]
Description=vLLM Inference Server (Qwen2.5-3B-GPTQ)
Documentation=https://docs.vllm.ai
After=network.target

[Service]
Type=simple
User=azureuser
Group=azureuser
WorkingDirectory=/home/azureuser

# Environment configuration
Environment="CUDA_VISIBLE_DEVICES=0"
Environment="PATH=/home/azureuser/vllm-venv/bin:/usr/local/cuda-12.4/bin:/usr/local/bin:/usr/bin:/bin"
Environment="LD_LIBRARY_PATH=/usr/local/cuda-12.4/lib64"

# vLLM serve command
# - GPTQ model (not AWQ - T4 incompatible)
# - dtype=half (not bfloat16 - T4 incompatible)
# - max-model-len=2048 (saves VRAM, sufficient for sentence analysis)
# - host=127.0.0.1 (localhost only, VNet access via private IP not needed for now)
# - gpu-memory-utilization=0.9 (leave 10% headroom)
ExecStart=/home/azureuser/vllm-venv/bin/vllm serve \
    /home/azureuser/models/Qwen2.5-3B-Instruct-GPTQ-Int4 \
    --dtype half \
    --max-model-len 2048 \
    --port 8001 \
    --host 127.0.0.1 \
    --gpu-memory-utilization 0.9

# Restart configuration
Restart=always
RestartSec=10
TimeoutStartSec=300
TimeoutStopSec=30

# Logging
StandardOutput=journal
StandardError=journal
SyslogIdentifier=vllm

[Install]
WantedBy=multi-user.target
```

2. Copy service file to VM and enable:

```bash
# Copy updated service file
scp infra/azure/vllm-vm/vllm.service azureuser@$VM_IP:/tmp/

# SSH to VM and install service
ssh azureuser@$VM_IP "sudo cp /tmp/vllm.service /etc/systemd/system/ && sudo systemctl daemon-reload && sudo systemctl enable vllm && sudo systemctl start vllm"

# Wait for model to load (takes 30-60 seconds)
sleep 60

# Check service status
ssh azureuser@$VM_IP "sudo systemctl status vllm"
```

3. Verify service is running and model is loaded by checking logs:

```bash
ssh azureuser@$VM_IP "sudo journalctl -u vllm --no-pager -n 50"
```

Look for: "Uvicorn running on http://127.0.0.1:8001" and model loading success messages.

1. Update `scripts/test-vllm-vm.sh` to include all verification tests:

```bash
#!/bin/bash
set -euo pipefail

# Inclusify vLLM VM Verification Script
# Usage: ./scripts/test-vllm-vm.sh [VM_IP]

# Get VM IP from argument or Azure CLI
VM_IP="${1:-}"
if [ -z "$VM_IP" ]; then
    echo "Discovering VM IP from Azure..."
    VM_IP=$(az network public-ip show --resource-group Group07 --name InclusifyModel-ip --query ipAddress -o tsv)
fi

if [ -z "$VM_IP" ]; then
    echo "ERROR: Could not determine VM IP. Pass as argument or ensure Azure CLI is configured."
    exit 1
fi

echo "=== Inclusify vLLM VM Verification ==="
echo "VM IP: $VM_IP"
echo ""

PASS=0
FAIL=0

# Test 1: Service status
echo "[1/4] Checking vLLM service status..."
if ssh -o ConnectTimeout=10 azureuser@$VM_IP "sudo systemctl is-active vllm" &>/dev/null; then
    echo "  PASS: vLLM service is active"
    ((PASS++))
else
    echo "  FAIL: vLLM service is not active"
    ((FAIL++))
fi

# Test 2: Health endpoint
echo "[2/4] Checking /v1/models endpoint..."
MODELS=$(ssh azureuser@$VM_IP "curl -s http://localhost:8001/v1/models" 2>/dev/null)
if echo "$MODELS" | grep -q '"data"'; then
    echo "  PASS: /v1/models returns model list"
    ((PASS++))
else
    echo "  FAIL: /v1/models did not return expected response"
    ((FAIL++))
fi

# Test 3: English inference
echo "[3/4] Testing English inference..."
ENGLISH_RESPONSE=$(ssh azureuser@$VM_IP 'curl -s http://localhost:8001/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d '"'"'{
    "model": "Qwen2.5-3B-Instruct-GPTQ-Int4",
    "messages": [{"role": "user", "content": "Hello, how are you?"}],
    "max_tokens": 50
  }'"'"'' 2>/dev/null)
if echo "$ENGLISH_RESPONSE" | grep -q '"choices"'; then
    echo "  PASS: English inference successful"
    ((PASS++))
else
    echo "  FAIL: English inference failed"
    echo "  Response: $ENGLISH_RESPONSE"
    ((FAIL++))
fi

# Test 4: Hebrew inference
echo "[4/4] Testing Hebrew inference..."
HEBREW_RESPONSE=$(ssh azureuser@$VM_IP 'curl -s http://localhost:8001/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d '"'"'{
    "model": "Qwen2.5-3B-Instruct-GPTQ-Int4",
    "messages": [{"role": "user", "content": "שלום, מה שלומך?"}],
    "max_tokens": 50
  }'"'"'' 2>/dev/null)
if echo "$HEBREW_RESPONSE" | grep -q '"choices"'; then
    echo "  PASS: Hebrew inference successful"
    ((PASS++))
else
    echo "  FAIL: Hebrew inference failed"
    echo "  Response: $HEBREW_RESPONSE"
    ((FAIL++))
fi

# Summary
echo ""
echo "=== Results ==="
echo "PASS: $PASS / $((PASS + FAIL))"
echo "FAIL: $FAIL / $((PASS + FAIL))"

if [ $FAIL -eq 0 ]; then
    echo ""
    echo "All tests passed! vLLM is ready for inference."
    exit 0
else
    echo ""
    echo "Some tests failed. Check vLLM logs: ssh azureuser@$VM_IP 'sudo journalctl -u vllm -n 100'"
    exit 1
fi
```

2. Make script executable: `chmod +x scripts/test-vllm-vm.sh`

3. Run verification: `./scripts/test-vllm-vm.sh`

4. Expected output: All 4 tests pass (service active, models endpoint, English inference, Hebrew inference).
